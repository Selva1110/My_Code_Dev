98 void XMC_I2C_CH_Init(XMC_USIC_CH_t *const channel, const XMC_I2C_CH_CONFIG_t *const config)
99 {
100  XMC_USIC_CH_Enable(channel);
101 
102  /* Data format configuration */
103  channel->SCTR = ((uint32_t)TRANSMISSION_MODE << (uint32_t)USIC_CH_SCTR_TRM_Pos) | /* Transmision mode */
104  ((uint32_t)WORDLENGTH << (uint32_t)USIC_CH_SCTR_WLE_Pos) | /* 8 data bits */
105  USIC_CH_SCTR_FLE_Msk | /* unlimited data flow */
106  USIC_CH_SCTR_SDIR_Msk | /* MSB shifted first */
107  USIC_CH_SCTR_PDL_Msk; /* Passive Data Level */
108 
109  XMC_I2C_CH_SetSlaveAddress(channel, config->address);
110  (void)XMC_I2C_CH_SetBaudrate(channel, config->baudrate);
111 
112  /* Enable transfer buffer */
113  channel->TCSR = ((uint32_t)SET_TDV << (uint32_t)USIC_CH_TCSR_TDEN_Pos) | USIC_CH_TCSR_TDSSM_Msk;
114 
115  /* Clear status flags */
116  channel->PSCR = 0xFFFFFFFF;
117 
118  /* Disable parity generation */
119  channel->CCR = 0x0U;
120 }
