70 void XMC_USIC_CH_Enable(XMC_USIC_CH_t *const channel)
71 {
72  XMC_ASSERT("XMC_USIC_CH_Enable: channel not valid", XMC_USIC_CHECK_CH(channel));
73 
74  if ((channel == XMC_USIC0_CH0) || (channel == XMC_USIC0_CH1))
75  {
76  XMC_USIC_Enable(XMC_USIC0);
77  }
78 #if defined(USIC1)
79  else if((channel == XMC_USIC1_CH0) || (channel == XMC_USIC1_CH1))
80  {
81  XMC_USIC_Enable(XMC_USIC1);
82  }
83 #endif
84 #if defined(USIC2)
85  else if((channel == XMC_USIC2_CH0) || (channel == XMC_USIC2_CH1))
86  {
87  XMC_USIC_Enable(XMC_USIC2);
88  }
89 #endif
90  else
91  {
92  XMC_ASSERT("USIC module not available", 0U/*Always*/);
93  }
94  /* USIC channel switched on*/
95  channel->KSCFG = (USIC_CH_KSCFG_MODEN_Msk | USIC_CH_KSCFG_BPMODEN_Msk);
96  while ((channel->KSCFG & USIC_CH_KSCFG_MODEN_Msk) == 0U)
97  {
98  /* Wait till the channel is enabled */
99  }
100 
101  /* Set USIC channel in IDLE mode */
102  channel->CCR &= (uint32_t)~USIC_CH_CCR_MODE_Msk;
103 }

110 XMC_USIC_CH_STATUS_t XMC_USIC_CH_SetBaudrate(XMC_USIC_CH_t *const channel, uint32_t rate, uint32_t oversampling)
111 {
112  XMC_USIC_CH_STATUS_t status;
113 
114  uint32_t peripheral_clock;
115 
116  uint32_t clock_divider;
117  uint32_t clock_divider_min;
118 
119  uint32_t pdiv;
120  uint32_t pdiv_int;
121  uint32_t pdiv_int_min;
122 
123  uint32_t pdiv_frac;
124  uint32_t pdiv_frac_min;
125 
126  /* The rate and peripheral clock are divided by 100 to be able to use only 32bit arithmetic */
127  if ((rate >= 100U) && (oversampling != 0U))
128  {
129  peripheral_clock = XMC_SCU_CLOCK_GetPeripheralClockFrequency() / 100U;
130  rate = rate / 100U;
131 
132  clock_divider_min = 1U;
133  pdiv_int_min = 1U;
134  pdiv_frac_min = 0x3ffU;
135 
136  for(clock_divider = 1023U; clock_divider > 0U; --clock_divider)
137  {
138  pdiv = ((peripheral_clock * clock_divider) / (rate * oversampling));
139  pdiv_int = pdiv >> 10U;
140  pdiv_frac = pdiv & 0x3ffU;
141 
142  if ((pdiv_int < 1024U) && (pdiv_frac < pdiv_frac_min))
143  {
144  pdiv_frac_min = pdiv_frac;
145  pdiv_int_min = pdiv_int;
146  clock_divider_min = clock_divider;
147  }
148  }
149 
150  channel->FDR = XMC_USIC_CH_BRG_CLOCK_DIVIDER_MODE_FRACTIONAL |
151  (clock_divider_min << USIC_CH_FDR_STEP_Pos);
152 
153  channel->BRG = (channel->BRG & ~(USIC_CH_BRG_DCTQ_Msk |
154  USIC_CH_BRG_PDIV_Msk |
155  USIC_CH_BRG_PCTQ_Msk |
156  USIC_CH_BRG_PPPEN_Msk)) |
157  ((oversampling - 1U) << USIC_CH_BRG_DCTQ_Pos) |
158  ((pdiv_int_min - 1U) << USIC_CH_BRG_PDIV_Pos);
159 
160  status = XMC_USIC_CH_STATUS_OK;
161  }
162  else
163  {
164  status = XMC_USIC_CH_STATUS_ERROR;
165  }
166 
167  return status;
168 }

170 void XMC_USIC_CH_TXFIFO_Configure(XMC_USIC_CH_t *const channel,
171  const uint32_t data_pointer,
172  const XMC_USIC_CH_FIFO_SIZE_t size,
173  const uint32_t limit)
174 {
175  /* Disable FIFO */
176  channel->TBCTR &= (uint32_t)~USIC_CH_TBCTR_SIZE_Msk;
177 
178  /* LOF = 0, A standard transmit buffer event occurs when the filling level equals the limit value and gets
179  * lower due to transmission of a data word
180  * STBTEN = 0, the trigger of the standard transmit buffer event is based on the transition of the fill level
181  * from equal to below the limit, not the fact being below
182  */
183  channel->TBCTR = (uint32_t)(channel->TBCTR & (uint32_t)~(USIC_CH_TBCTR_LIMIT_Msk |
184  USIC_CH_TBCTR_DPTR_Msk |
185  USIC_CH_TBCTR_SIZE_Msk)) |
186  (uint32_t)((limit << USIC_CH_TBCTR_LIMIT_Pos) |
187  (data_pointer << USIC_CH_TBCTR_DPTR_Pos) |
188  ((uint32_t)size << USIC_CH_TBCTR_SIZE_Pos));
189 }

202  */
203  channel->RBCTR = (uint32_t)((channel->RBCTR & (uint32_t)~(USIC_CH_RBCTR_LIMIT_Msk |
