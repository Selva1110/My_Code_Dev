124 /* Initializes the selected ERU_ETLx channel with the config structure. */
125 void XMC_ERU_ETL_Init(XMC_ERU_t *const eru,
126  const uint8_t channel,
127  const XMC_ERU_ETL_CONFIG_t *const config)
128 {
129  XMC_ASSERT("XMC_ERU_ETL_Init:Invalid Module Pointer", XMC_ERU_CHECK_MODULE_PTR(eru));
130  XMC_ASSERT("XMC_ERU_ETL_Init:Invalid Channel Number", (channel < 4U));
131 
132  XMC_ERU_Enable(eru);
133 
134  eru->EXISEL = (eru->EXISEL &
135  ~((uint32_t)(ERU_EXISEL_EXS0A_Msk | ERU_EXISEL_EXS0B_Msk) << (channel * ERU_EXISEL_BITSIZE))) |
136  (config->input << (channel * (uint32_t)ERU_EXISEL_BITSIZE));
137 
138  eru->EXICON[channel] = config->raw;
139 }
