602 /* API to configure a slice trigger event */
603 void XMC_CCU4_SLICE_ConfigureEvent(XMC_CCU4_SLICE_t *const slice,
604  const XMC_CCU4_SLICE_EVENT_t event,
605  const XMC_CCU4_SLICE_EVENT_CONFIG_t *const config)
606 {
607  uint32_t ins;
608  uint8_t pos;
609  uint8_t offset;
610 
611  XMC_ASSERT("XMC_CCU4_SLICE_ConfigureEvent:Invalid Slice Pointer", XMC_CCU4_CHECK_SLICE_PTR(slice));
612  XMC_ASSERT("XMC_CCU4_SLICE_ConfigureEvent:Invalid Event ID", XMC_CCU4_SLICE_CHECK_EVENT_ID(event));
613  XMC_ASSERT("XMC_CCU4_SLICE_ConfigureEvent:Invalid Input", XMC_CCU4_SLICE_CHECK_INPUT(config->mapped_input));
614  XMC_ASSERT("XMC_CCU4_SLICE_ConfigureEvent:Invalid Edge Sensitivity",
615  XMC_CCU4_SLICE_CHECK_EDGE_SENSITIVITY(config->edge));
616  XMC_ASSERT("XMC_CCU4_SLICE_ConfigureEvent:Invalid Level Sensitivity",
617  ((config->level == XMC_CCU4_SLICE_EVENT_LEVEL_SENSITIVITY_ACTIVE_HIGH) ||\
618  (config->level == XMC_CCU4_SLICE_EVENT_LEVEL_SENSITIVITY_ACTIVE_LOW)));
619  XMC_ASSERT("XMC_CCU4_SLICE_ConfigureEvent:Invalid Debounce Period",
620  XMC_CCU4_SLICE_CHECK_EVENT_FILTER(config->duration));
621 
622  ins = slice->INS;
623 
624  /* Calculate offset with reference to event */
625  offset = ((uint8_t) event) - 1U;
626 
627  /* First, configure the edge sensitivity */
628  pos = ((uint8_t) CCU4_CC4_INS_EV0EM_Pos) + (uint8_t)(offset << 1U);
629  ins &= ~(((uint32_t) XMC_CCU4_SLICE_EVENT_EDGE_CONFIG_MASK) << pos);
630  ins |= ((uint32_t) config->edge) << pos;
631 
632  /* Next, the level */
633  pos = ((uint8_t) CCU4_CC4_INS_EV0LM_Pos) + offset;
634  ins &= ~(((uint32_t) XMC_CCU4_SLICE_EVENT_LEVEL_CONFIG_MASK) << pos);
635  ins |= ((uint32_t) config->level) << pos;
636 
637  /* Next, the debounce filter */
638  pos = ((uint8_t) CCU4_CC4_INS_LPF0M_Pos) + (uint8_t)(offset << 1U);
639  ins &= ~(((uint32_t) XMC_CCU4_SLICE_EVENT_FILTER_CONFIG_MASK) << pos);
640  ins |= ((uint32_t) config->duration) << pos;
641 
642  /* Finally the input */
643  pos = ((uint8_t) CCU4_CC4_INS_EV0IS_Pos) + (uint8_t)(offset << 2U);
644  ins &= ~(((uint32_t) XMC_CCU4_SLICE_EVENT_INPUT_CONFIG_MASK) << pos);
645  ins |= ((uint32_t) config->mapped_input) << pos;
646 
647  slice->INS = ins;
648 }
