433 /* A utility routine which updates the fractional dividers in steps */
434 static void XMC_SCU_CLOCK_lDivUpdate(uint32_t idiv, uint32_t fdiv)
435 {
436  /* Find out current and target value of idiv */
437  uint32_t curr_idiv;
438 
439  /* Take a snapshot of value already programmed into IDIV */
440  curr_idiv = (SCU_CLK->CLKCR & SCU_CLK_CLKCR_IDIV_Msk) >> SCU_CLK_CLKCR_IDIV_Pos;
441 
442  SCU_CLK->CLKCR = (SCU_CLK->CLKCR & (uint32_t)~(SCU_CLK_CLKCR_FDIV_Msk | SCU_CLK_CLKCR_CNTADJ_Pos)) |
443  (uint32_t)(fdiv << SCU_CLK_CLKCR_FDIV_Pos) |
444  (uint32_t)(1023UL <<SCU_CLK_CLKCR_CNTADJ_Pos);
445 
446  while ((SCU_CLK->CLKCR)& SCU_CLK_CLKCR_VDDC2LOW_Msk)
447  {
448  /* Spin until the core supply stabilizes */
449  }
450 
451  if(curr_idiv <= idiv)
452  {
453  /* Requested IDIV is greater than currently programmed IDIV. So downscale the frequency */
454  XMC_SCU_CLOCK_lFrequencyDownScaling(curr_idiv, idiv);
455  }
456  else
457  {
458  /* Requested IDIV is lower than currently programmed IDIV. So upscale the frequency */
459  XMC_SCU_CLOCK_lFrequencyUpScaling(curr_idiv, idiv);
460  }
461 
462  SCU_CLK->CLKCR = (SCU_CLK->CLKCR & (uint32_t)~(SCU_CLK_CLKCR_IDIV_Msk | SCU_CLK_CLKCR_CNTADJ_Msk)) |
463  (uint32_t)(idiv << SCU_CLK_CLKCR_IDIV_Pos) | (uint32_t)(1023UL << SCU_CLK_CLKCR_CNTADJ_Pos);
464 
465  while ((SCU_CLK->CLKCR) & SCU_CLK_CLKCR_VDDC2LOW_Msk)
466  {
467  /* Wait voltage suply stabilization */
468  }
469 
470 }
