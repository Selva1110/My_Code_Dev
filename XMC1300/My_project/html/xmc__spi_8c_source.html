72 /* Initializes the selected SPI channel with the config structure. */
73 void XMC_SPI_CH_Init(XMC_USIC_CH_t *const channel, const XMC_SPI_CH_CONFIG_t *const config)
74 {
75  XMC_USIC_CH_Enable(channel);
76 
77  if(config->bus_mode == XMC_SPI_CH_BUS_MODE_MASTER)
78  {
79  /* Configure baud rate */
80  (void)XMC_USIC_CH_SetBaudrate(channel, config->baudrate, XMC_SPI_CH_OVERSAMPLING);
81  }
82 
83  /* Configuration of USIC Shift Control */
84  /* Transmission Mode (TRM) = 1 */
85  /* Passive Data Level (PDL) = 1 */
86  channel->SCTR = USIC_CH_SCTR_PDL_Msk |
87  (0x1UL << USIC_CH_SCTR_TRM_Pos) |
88  (0x3fUL << USIC_CH_SCTR_FLE_Pos)|
89  (0x7UL << USIC_CH_SCTR_WLE_Pos);
90 
91  /* Configuration of USIC Transmit Control/Status Register */
92  /* TBUF Data Enable (TDEN) = 1 */
93  /* TBUF Data Single Shot Mode (TDSSM) = 1 */
94  channel->TCSR = (uint32_t)(USIC_CH_TCSR_HPCMD_Msk |
95  (0x01UL << USIC_CH_TCSR_TDEN_Pos) |
96  USIC_CH_TCSR_TDSSM_Msk);
97 
98  if(config->bus_mode == XMC_SPI_CH_BUS_MODE_MASTER)
99  {
100  /* Configuration of Protocol Control Register */
101  channel->PCR_SSCMode = (uint32_t)(USIC_CH_PCR_SSCMode_MSLSEN_Msk |
102  USIC_CH_PCR_SSCMode_SELCTR_Msk |
103  (uint32_t)config->selo_inversion |
104  USIC_CH_PCR_SSCMode_FEM_Msk);
105  }
106 
107  /* Clear protocol status */
108  channel->PSCR = 0xFFFFFFFFUL;
109 
110  /* Set parity settings */
111  channel->CCR = (uint32_t)config->parity_mode;
112 }

147 /* Puts the data into FIFO if FIFO mode is enabled or else into standard buffers, by setting the proper mode. */
148 void XMC_SPI_CH_Transmit(XMC_USIC_CH_t *const channel, const uint16_t data, const XMC_SPI_CH_MODE_t mode)
149 {
150 
151  channel->CCR = (channel->CCR & (uint32_t)(~USIC_CH_CCR_HPCEN_Msk)) |
152  (((uint32_t) mode << USIC_CH_CCR_HPCEN_Pos) & (uint32_t)USIC_CH_CCR_HPCEN_Msk);
153 
154 
155  /* Check FIFO size */
156  if ((channel->TBCTR & USIC_CH_TBCTR_SIZE_Msk) == 0U)
157  {
158  while((uint32_t)XMC_USIC_CH_GetTransmitBufferStatus(channel) & (uint32_t)XMC_USIC_CH_TBUF_STATUS_BUSY)
159  {
160  }
161 
162  XMC_SPI_CH_ClearStatusFlag(channel, (uint32_t)XMC_SPI_CH_STATUS_FLAG_TRANSMIT_BUFFER_INDICATION);
163 
164  channel->TBUF[mode] = data;
165  }
166  else
167  {
168  channel->IN[mode] = data;

190 /* Configures the inter word delay by setting PCR.PCTQ1, PCR.DCTQ1 bit fields. */
191 void XMC_SPI_CH_SetInterwordDelay(XMC_USIC_CH_t *const channel,uint32_t tinterword_delay_us)
192 {
193  uint32_t peripheral_clock;
194  uint32_t pdiv;
195  uint32_t step;
196  uint32_t fFD;
197  uint32_t fpdiv;
198  uint32_t divider_factor1 = 0U;
199  uint32_t divider_factor2 = 32U;
200  uint32_t divider_factor1_int = 0U;
201  uint32_t divider_factor1_int_min = 4U;
202  uint32_t divider_factor1_frac_min =100U;
203  uint32_t divider_factor1_frac = 0U;
204  uint32_t divider_factor2_temp = 0U;
205  peripheral_clock = XMC_SCU_CLOCK_GetPeripheralClockFrequency();
206  pdiv = (uint32_t)(channel->BRG & USIC_CH_BRG_PDIV_Msk) >> USIC_CH_BRG_PDIV_Pos;
207  step = (uint32_t)(channel->FDR & USIC_CH_FDR_STEP_Msk) >> USIC_CH_FDR_STEP_Pos;
208  fFD = (uint32_t)((peripheral_clock >> 10U) * step);
209  fpdiv= fFD/(1U+pdiv);
210 
211  if(tinterword_delay_us < (128000000/fpdiv))
212  {
213  for(divider_factor2_temp = 32U; divider_factor2_temp > 0U; --divider_factor2_temp)
214  {
215 
216  divider_factor1 = (tinterword_delay_us*fpdiv)/(divider_factor2_temp*10000);
217  divider_factor1_frac = divider_factor1%100U;
218 
219  if(divider_factor1_frac > 50)
220  {
221  divider_factor1_int = (divider_factor1/100U)+1;
222  divider_factor1_frac = (divider_factor1_int*100)-divider_factor1;
223  }
224  else
225  {
226  divider_factor1_int = (divider_factor1/100U);
227  }
228 
229  if ((divider_factor1_int < 5U) && (divider_factor1_int > 0) && (divider_factor1_frac < divider_factor1_frac_min))
230  {
231  divider_factor1_frac_min = divider_factor1_frac;
232  divider_factor1_int_min = divider_factor1_int;
233  divider_factor2= divider_factor2_temp;
234  }
235  }
236  }
237 
238  channel->PCR_SSCMode = (uint32_t)((channel->PCR_SSCMode) & (~(USIC_CH_PCR_SSCMode_DCTQ1_Msk |
239  USIC_CH_PCR_SSCMode_PCTQ1_Msk |
240  USIC_CH_PCR_SSCMode_CTQSEL1_Msk))) |
241  (((divider_factor1_int_min - 1) << USIC_CH_PCR_SSCMode_PCTQ1_Pos) & (uint32_t)USIC_CH_PCR_SSCMode_PCTQ1_Msk) |
242  (((divider_factor2 - 1 ) << USIC_CH_PCR_SSCMode_DCTQ1_Pos) & (uint32_t)USIC_CH_PCR_SSCMode_DCTQ1_Msk);
243 }
