86 void XMC_UART_CH_Init(XMC_USIC_CH_t *channel, const XMC_UART_CH_CONFIG_t *const config)
87 {
88  uint32_t oversampling = XMC_UART_CH_OVERSAMPLING;
89 
90  /* USIC channel switched on*/
91  XMC_USIC_CH_Enable(channel);
92 
93  if(config->oversampling != 0U)
94  {
95  oversampling = (uint32_t)config->oversampling;
96  }
97 
98  /* Configure baud rate */
99  (void)XMC_USIC_CH_SetBaudrate(channel, config->baudrate, oversampling);
100 
101  /* Configure frame format
102  * Configure the number of stop bits
103  * Pulse length is set to 0 to have standard UART signaling,
104  * i.e. the 0 level is signaled during the complete bit time
105  * Sampling point set equal to the half of the oversampling period
106  * Enable Sample Majority Decision */
107  channel->PCR_ASCMode = (uint32_t)(((config->stop_bits - 1UL) << USIC_CH_PCR_ASCMode_STPB_Pos) |
108  (((oversampling >> 1UL) + 1UL) << USIC_CH_PCR_ASCMode_SP_Pos) |
109  USIC_CH_PCR_ASCMode_SMD_Msk);
110 
111  /* Set passive data level, high
112  Set word length. Data bits - 1
113  If frame length is > 0, frame_lemgth-1; else, FLE = WLE (Data bits - 1)
114  Transmission Mode: The shift control signal is considered active if it
115  is at 1-level. This is the setting to be programmed to allow data transfers */
116  channel->SCTR = (uint32_t)((((uint32_t)config->data_bits - 1UL) << USIC_CH_SCTR_WLE_Pos) |
117  ((0x1UL << USIC_CH_SCTR_TRM_Pos) | USIC_CH_SCTR_PDL_Msk));
118 
119  if (config->frame_length != 0U)
120  {
121  channel->SCTR |= (uint32_t)(((uint32_t)config->frame_length - 1UL) << USIC_CH_SCTR_FLE_Pos);
122  }
123  else
124  {
125  channel->SCTR |= (uint32_t)(((uint32_t)config->data_bits - 1UL) << USIC_CH_SCTR_FLE_Pos);
126  }
127 
128  /* Enable transfer buffer */
129  channel->TCSR = (0x1UL << USIC_CH_TCSR_TDEN_Pos) |
130  USIC_CH_TCSR_TDSSM_Msk;
131 
132  /* Clear protocol status */
133  channel->PSCR = 0xFFFFFFFFUL;
134 
135  /* Set parity settings */
136  channel->CCR = (uint32_t)config->parity_mode;
137 }
