249  * the given time tick value is within supported range.

251 TIMER_STATUS_t TIMER_SetTimeInterval(TIMER_t *const handle_ptr, uint32_t time_interval)
252 {
253  TIMER_STATUS_t status;
254  uint32_t lfrequency;
255  uint32_t lprescaler;
256 
257  XMC_ASSERT("TIMER_SetTimeInterval:handle_ptr NULL" , (handle_ptr != NULL));
258 
259  status = TIMER_STATUS_FAILURE;
260 
261  if (false == TIMER_GetTimerStatus(handle_ptr))
262  {
263  /* check for time_interval range */
264  if ((time_interval >= handle_ptr->timer_min_value_us) && (time_interval <= handle_ptr->timer_max_value_us))
265  {
266  /* Initialize the prescaler */
267  lprescaler = 0U;
268  while (time_interval > (handle_ptr->timer_max_value_us >> (TIMER_PRESCALER_MAX - lprescaler)))
269  {
270  lprescaler++;
271  }
272 #ifdef TIMER_CCU4_USED
273  if (TIMER_MODULE_CCU4 == handle_ptr->timer_module)
274  {
275  lfrequency = handle_ptr->global_ccu4_handler->module_frequency;
276  handle_ptr->ccu4_slice_config_ptr->prescaler_initval = lprescaler;
277  /* Calculate the period register for the required time_interval value */
278  handle_ptr->period_value = (uint16_t)((((uint64_t)time_interval * lfrequency) >> \
279  handle_ptr->ccu4_slice_config_ptr->prescaler_initval) / \
280  TIMER_RESOLUTION_SEC_TO_MICRO);
281  /* Actual timer period values is Period_reg_val+1U */
282  if (handle_ptr->period_value > TIMER_PERIOD_MIN)
283  {
284  (handle_ptr->period_value)--;
285  }
286  /* Update the prescaler */
287  XMC_CCU4_SLICE_SetPrescaler(handle_ptr->ccu4_slice_ptr, handle_ptr->ccu4_slice_config_ptr->prescaler_initval);
288  /* update period, compare and prescaler values */
289  TIMER_CCU4_lShadowTransfer(handle_ptr);
290  /* Update the status */
291  status = TIMER_STATUS_SUCCESS;
292  }
293 #endif
294 
295 #ifdef TIMER_CCU8_USED
296  if (TIMER_MODULE_CCU8 == handle_ptr->timer_module)
297  {
298  handle_ptr->ccu8_slice_config_ptr->prescaler_initval = lprescaler;
299  lfrequency = handle_ptr->global_ccu8_handler->module_frequency;
300  /* Calculate the period register for the required time_interval value */
301  handle_ptr->period_value = (uint16_t)((((uint64_t)time_interval * lfrequency) >> \
302  handle_ptr->ccu8_slice_config_ptr->prescaler_initval) / \
303  TIMER_RESOLUTION_SEC_TO_MICRO);
304  /* Actual timer period values is Period_reg_val+1U */
