147 {
148  ASSERT(hw == ADC);
149 
150  uint16_t calib_reg;
151 
152  calib_reg = ADC_CALIB_BIAS_CAL((*(uint32_t *)ADC_FUSES_BIASCAL_ADDR >> ADC_FUSES_BIASCAL_Pos))
153  | ADC_CALIB_LINEARITY_CAL((*(uint64_t *)ADC_FUSES_LINEARITY_0_ADDR >> ADC_FUSES_LINEARITY_0_Pos));
154 
155  hri_adc_wait_for_sync(hw);
156  if (hri_adc_get_CTRLA_reg(hw, ADC_CTRLA_ENABLE)) {
157  hri_adc_write_CTRLA_reg(hw, 0);
158  hri_adc_wait_for_sync(hw);
159  }
160  hri_adc_write_CTRLA_reg(hw, ADC_CTRLA_SWRST);
161  hri_adc_wait_for_sync(hw);
162 
163  hri_adc_write_CALIB_reg(hw, calib_reg);
164  hri_adc_write_REFCTRL_reg(hw, _adcs[i].ref_ctrl);
165  hri_adc_write_AVGCTRL_reg(hw, _adcs[i].avg_ctrl);
166  hri_adc_write_SAMPCTRL_reg(hw, _adcs[i].samp_ctrl);
167  hri_adc_write_EVCTRL_reg(hw, _adcs[i].ev_ctrl);
168  hri_adc_write_GAINCORR_reg(hw, _adcs[i].gain_corr);
169  hri_adc_write_OFFSETCORR_reg(hw, _adcs[i].offset_corr);
170  hri_adc_write_DBGCTRL_reg(hw, _adcs[i].dbg_ctrl);
171  hri_adc_write_CTRLB_reg(hw, _adcs[i].ctrl_b);
172  hri_adc_write_INPUTCTRL_reg(hw, _adcs[i].input_ctrl);
173  hri_adc_write_WINCTRL_reg(hw, _adcs[i].win_ctrl);
174  hri_adc_write_WINLT_reg(hw, _adcs[i].win_lt);
175  hri_adc_write_WINUT_reg(hw, _adcs[i].win_ut);
176  hri_adc_write_CTRLA_reg(hw, _adcs[i].ctrl_a);

178  return ERR_NONE;
