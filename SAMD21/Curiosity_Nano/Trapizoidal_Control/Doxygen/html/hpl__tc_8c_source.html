147 int32_t _tc_timer_init(struct _timer_device *const device, void *const hw)
148 {
149  int8_t i = get_tc_index(hw);
150 
151  device->hw = hw;
152  ASSERT(ARRAY_SIZE(_tcs));
153 
154  hri_tc_wait_for_sync(hw);
155  if (hri_tc_get_CTRLA_reg(hw, TC_CTRLA_ENABLE)) {
156  hri_tc_write_CTRLA_reg(hw, 0);
157  hri_tc_wait_for_sync(hw);
158  }
159  hri_tc_write_CTRLA_reg(hw, TC_CTRLA_SWRST);
160  hri_tc_wait_for_sync(hw);
161 
162  hri_tc_write_CTRLA_reg(hw, _tcs[i].ctrl_a);
163  hri_tc_write_DBGCTRL_reg(hw, _tcs[i].dbg_ctrl);
164  hri_tc_write_EVCTRL_reg(hw, _tcs[i].event_ctrl);
165 
166  if ((_tcs[i].ctrl_a & TC_CTRLA_MODE_Msk) == TC_CTRLA_MODE_COUNT32) {
167  hri_tccount32_write_CC_reg(hw, 0, _tcs[i].cc0);
168  hri_tccount32_write_CC_reg(hw, 1, _tcs[i].cc1);
169  } else if ((_tcs[i].ctrl_a & TC_CTRLA_MODE_Msk) == TC_CTRLA_MODE_COUNT16) {
170  hri_tccount16_write_CC_reg(hw, 0, (hri_tccount16_cc_reg_t)_tcs[i].cc0);
171  hri_tccount16_write_CC_reg(hw, 1, (hri_tccount16_cc_reg_t)_tcs[i].cc1);
172  } else if ((_tcs[i].ctrl_a & TC_CTRLA_MODE_Msk) == TC_CTRLA_MODE_COUNT8) {
173  hri_tccount8_write_CC_reg(hw, 0, (hri_tccount8_cc_reg_t)_tcs[i].cc0);
174  hri_tccount8_write_CC_reg(hw, 1, (hri_tccount8_cc_reg_t)_tcs[i].cc1);
175  hri_tccount8_write_PER_reg(hw, _tcs[i].per);
176  }
177  hri_tc_set_INTEN_OVF_bit(hw);
178 
179  _tc_init_irq_param(hw, (void *)device);
180  NVIC_DisableIRQ((IRQn_Type)((uint8_t)TC_IRQ_BASE_INDEX + tc_get_hardware_index(hw)));
181  NVIC_ClearPendingIRQ((IRQn_Type)((uint8_t)TC_IRQ_BASE_INDEX + tc_get_hardware_index(hw)));
182  NVIC_EnableIRQ((IRQn_Type)((uint8_t)TC_IRQ_BASE_INDEX + tc_get_hardware_index(hw)));
183 
184  return ERR_NONE;
185 }
