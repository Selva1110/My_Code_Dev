589 {

591 
592  if (!hri_sercomusart_is_syncing(hw, SERCOM_USART_SYNCBUSY_SWRST)) {

594  if (hri_sercomusart_get_CTRLA_reg(hw, SERCOM_USART_CTRLA_ENABLE)) {
595  hri_sercomusart_clear_CTRLA_ENABLE_bit(hw);
596  hri_sercomusart_wait_for_sync(hw, SERCOM_USART_SYNCBUSY_ENABLE);
597  }
598  hri_sercomusart_write_CTRLA_reg(hw, SERCOM_USART_CTRLA_SWRST | mode);

601 
602  hri_sercomusart_write_CTRLA_reg(hw, _usarts[i].ctrl_a);
603  hri_sercomusart_write_CTRLB_reg(hw, _usarts[i].ctrl_b);
604  if ((_usarts[i].ctrl_a & SERCOM_USART_CTRLA_SAMPR(0x1)) || (_usarts[i].ctrl_a & SERCOM_USART_CTRLA_SAMPR(0x3))) {
605  ((Sercom *)hw)->USART.BAUD.FRAC.BAUD = _usarts[i].baud;
606  ((Sercom *)hw)->USART.BAUD.FRAC.FP = _usarts[i].fractional;
607  } else {
608  hri_sercomusart_write_BAUD_reg(hw, _usarts[i].baud);
609  }
610 
611  hri_sercomusart_write_RXPL_reg(hw, _usarts[i].rxpl);

640  const enum usart_baud_rate_mode mode, const uint8_t fraction)
641 {
642  if (USART_BAUDRATE_ASYNCH_ARITHMETIC == mode) {

644  }
645 
646  if (USART_BAUDRATE_ASYNCH_FRACTIONAL == mode) {
647  return clock_rate / baud / samples + SERCOM_USART_BAUD_FRACFP_FP(fraction);
648  }
649

651  return clock_rate / baud / 2 - 1;
652  }
653 
654  return 0;

932 static inline int32_t _sercom_i2c_sync_analyse_flags(void *const hw, uint32_t flags, struct _i2c_m_msg *const msg)
933 {
934  int sclsm = hri_sercomi2cm_get_CTRLA_SCLSM_bit(hw);
935  uint16_t status = hri_sercomi2cm_read_STATUS_reg(hw);
936 
937  if (flags & MB_FLAG) {
938  /* tx error */
939  if (status & SERCOM_I2CM_STATUS_ARBLOST) {
940  hri_sercomi2cm_clear_interrupt_MB_bit(hw);
941  msg->flags |= I2C_M_FAIL;
942  msg->flags &= ~I2C_M_BUSY;
943 
944  if (status & SERCOM_I2CM_STATUS_BUSERR) {
945  return I2C_ERR_BUS;
946  }
947 
948  return I2C_ERR_BAD_ADDRESS;
949  } else {
950  if (status & SERCOM_I2CM_STATUS_RXNACK) {
951 
952  /* Slave rejects to receive more data */
953  if (msg->len > 0) {
954  msg->flags |= I2C_M_FAIL;
955  }
956 
957  if (msg->flags & I2C_M_STOP) {
958  _sercom_i2c_send_stop(hw);
959  }
960 
961  msg->flags &= ~I2C_M_BUSY;
962 
963  return I2C_NACK;
964  }
965 
966  if (msg->flags & I2C_M_TEN) {
967  hri_sercomi2cm_write_ADDR_reg(hw,
968  ((((msg->addr & TEN_ADDR_MASK) >> 8) | TEN_ADDR_FRAME) << 1) | I2C_M_RD
969  | (hri_sercomi2cm_read_ADDR_reg(hw) & SERCOM_I2CM_ADDR_HS));
970  msg->flags &= ~I2C_M_TEN;
971 
972  return I2C_OK;
973  }
974 
975  if (msg->len == 0) {
976  if (msg->flags & I2C_M_STOP) {
977  _sercom_i2c_send_stop(hw);
978  }
979 
980  msg->flags &= ~I2C_M_BUSY;
981  } else {
982  hri_sercomi2cm_write_DATA_reg(hw, *msg->buffer);
983  msg->buffer++;
984  msg->len--;
985  }
986 
987  return I2C_OK;
988  }
989  } else if (flags & SB_FLAG) {
990  if ((msg->len) && !(status & SERCOM_I2CM_STATUS_RXNACK)) {
991  msg->len--;
992 
993  /* last byte, send nack */
994  if ((msg->len == 0 && !sclsm) || (msg->len == 1 && sclsm)) {
995  hri_sercomi2cm_set_CTRLB_ACKACT_bit(hw);
996  }
997 
998  if (msg->len == 0) {
999  if (msg->flags & I2C_M_STOP) {
1000  hri_sercomi2cm_clear_CTRLB_SMEN_bit(hw);
1001  _sercom_i2c_send_stop(hw);
1002  }
1003 
1004  msg->flags &= ~I2C_M_BUSY;
1005  }
1006 
1007  /* Accessing DATA.DATA auto-triggers I2C bus operations.
1008  * The operation performed depends on the state of
1009  * CTRLB.ACKACT, CTRLB.SMEN
1010  **/
1011  *msg->buffer++ = hri_sercomi2cm_read_DATA_reg(hw);
1012  } else {
1013  hri_sercomi2cm_clear_interrupt_SB_bit(hw);
1014  return I2C_NACK;
1015  }
1016 
1017  hri_sercomi2cm_clear_interrupt_SB_bit(hw);
1018  }
1019 
1020  return I2C_OK;
1021 }
