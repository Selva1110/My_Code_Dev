382 {

385 
386  if (adp_ext_request_data(&response) == true) {

390  if (start_toggle == 1) {

394  state_run = 0;

403  speedpi.kp = kp_ref;
404  break;
405  case STREAM_ID_KI_IN:
406  ki_ref = (uint16_t)((response.data[1] << 8) + response.data[0]);

408  break;

410  ramp_speedperms = (uint16_t)((response.data[1] << 8) + response.data[0]);

412  case STREAM_ID_DIRECTION_IN:

422  }
