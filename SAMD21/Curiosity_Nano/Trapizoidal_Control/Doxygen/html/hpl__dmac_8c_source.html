162 {

165 
166  if (hri_dmacdescriptor_get_BTCTRL_DSTINC_bit(&_descriptor_section[channel])) {
167  hri_dmacdescriptor_write_DSTADDR_reg(&_descriptor_section[channel], address + amount * (1 << beat_size));

174  }

210 static inline void _dmac_handler(void)
211 {
212  uint8_t channel = hri_dmac_read_INTPEND_ID_bf(DMAC);
213  uint8_t current_channel = hri_dmac_read_CHID_reg(DMAC);
214  uint8_t flag_status;
215  struct _dma_resource *tmp_resource = &_resources[channel];
216 
217  hri_dmac_write_CHID_reg(DMAC, channel);
218  flag_status = hri_dmac_get_CHINTFLAG_reg(DMAC, DMAC_CHINTFLAG_MASK);
219  hri_dmac_write_CHID_reg(DMAC, current_channel);
220 
221  if (flag_status & DMAC_CHINTFLAG_TERR) {
222  hri_dmac_clear_CHINTFLAG_TERR_bit(DMAC);
223  tmp_resource->dma_cb.error(tmp_resource);
224  } else if (flag_status & DMAC_CHINTFLAG_TCMPL) {
225  hri_dmac_clear_CHINTFLAG_TCMPL_bit(DMAC);
226  tmp_resource->dma_cb.transfer_done(tmp_resource);
227  }
228 }
