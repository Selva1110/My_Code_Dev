78 static bool adp_add_receive_byte(uint8_t data)
79 {
80  static uint8_t message_id;
81 
82  if ((rx_state == RX_STATE_GOT_SYMBOL) && (data != ADP_TOKEN)) {
83  /* Abort packet reception, new packet incoming */
84  rx_state = RX_STATE_WAIT_LENGTH_LSB;
85  }
86 
87  switch (rx_state) {
88  case RX_STATE_IDLE:
89  packet_received = false;
90  last_received_message_id = 0xFF;
91  /* We are waiting for a new packet. */
92  if (data != ADP_TOKEN) {
93  return false;
94  }
95  /* Got start symbol, wait for message ID */
96  rx_state = RX_STATE_WAIT_MSG_ID;
97  return false;
98 
99  case RX_STATE_WAIT_MSG_ID:
100  if (data == ADP_TOKEN) {
101  /* Restart. Don't change state. Wait for new message ID */
102  return false;
103  }
104  message_id = data;
105  rx_state = RX_STATE_WAIT_LENGTH_LSB;
106  return false;
107 
108  case RX_STATE_WAIT_LENGTH_LSB:
109  if (data == ADP_TOKEN) {
110  if (prev_data != ADP_TOKEN) {
111  prev_data = ADP_TOKEN;
112  return false;
113  }
114  }
115  length_received = data;
116  rx_state = RX_STATE_WAIT_LENGTH_MSB;
117  prev_data = 0;
118  return false;
119 
120  case RX_STATE_WAIT_LENGTH_MSB:
121  if (data == ADP_TOKEN) {
122  if (prev_data != ADP_TOKEN) {
123  prev_data = ADP_TOKEN;
124  return false;
125  }
126  }
127  length_received += (uint16_t)data << 8;
128  prev_data = 0;
129  /* Got valid length, do we expect data? */
130  if (length_received == 0) {
131  /* No data here, wait for next packet */
132  rx_state = RX_STATE_IDLE;
133  packet_received = true;
134  last_received_message_id = message_id;
135  return false;
136  }
137 
138  /* Wait for packet data */
139  bytes_received = 0;
140  rx_state = RX_STATE_GET_DATA;
141  return false;
142 
143  case RX_STATE_GET_DATA:
144  case RX_STATE_GOT_SYMBOL:
145  if ((data == ADP_TOKEN) && (rx_state == RX_STATE_GET_DATA)) {
146  rx_state = RX_STATE_GOT_SYMBOL;
147  return false;
148  }
149  /* Add new data to rx buffer */
150  bytes_received++;
151  /* Are we done yet? */
152  if (length_received == bytes_received) {
153  /* Yes we are! */
154  packet_received = true;
155  rx_state = RX_STATE_IDLE;
156  last_received_message_id = message_id;
157  return true;
158  }
159  /* Not done yet.. keep on receiving */
160  rx_state = RX_STATE_GET_DATA;
161  return true;
162  }
163  return false;
164 }
